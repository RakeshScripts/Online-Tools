    <script>
        async function convertWebpToJpg() {
            const input = document.getElementById('webpInput');
            const status = document.getElementById('status');
            const result = document.getElementById('result');

            if (!input.files[0]) {
                alert('Please select a WEBP file first!');
                return;
            }

            // Show status
            status.style.display = 'block';
            result.innerHTML = '';

            try {
                const file = input.files[0];
                const imgURL = URL.createObjectURL(file);
                const img = new Image();

                img.onload = function() {
                    // Create canvas
                    const canvas = document.createElement('canvas');
                    canvas.width = img.width;
                    canvas.height = img.height;
                    const ctx = canvas.getContext('2d');
                    ctx.drawImage(img, 0, 0);

                    // Convert to JPG with quality setting
                    const jpgData = canvas.toDataURL('image/jpeg', 0.95);

                    // Display result
                    result.innerHTML = `
                        <p>Conversion successful!</p>
                        <img src="${jpgData}" alt="Converted JPG image">
                        <a href="${jpgData}" download="converted-image.jpg" class="download-btn">Download JPG</a>
                    `;

                    // Clean up
                    status.style.display = 'none';
                    URL.revokeObjectURL(imgURL);
                };

                img.onerror = function() {
                    throw new Error('Error loading image');
                };

                img.src = imgURL;
            } catch (error) {
                status.textContent = 'Error during conversion. Please try again.';
                status.style.color = '#c0392b';
                setTimeout(() => {
                    status.style.display = 'none';
                    status.style.color = 'inherit';
                }, 3000);
            }
        }
    </script>
